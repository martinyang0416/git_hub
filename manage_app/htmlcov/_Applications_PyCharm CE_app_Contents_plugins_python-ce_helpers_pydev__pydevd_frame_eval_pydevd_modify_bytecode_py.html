11def _add_attr_values_from_insert_to_original(original_code, insert_code, insert_code_list, attribute_name, op_list): 

12 """ 

13 This function appends values of the attribute `attribute_name` of the inserted code to the original values, 

14 and changes indexes inside inserted code. If some bytecode instruction in the inserted code used to call argument 

15 number i, after modification it calls argument n + i, where n - length of the values in the original code. 

16 So it helps to avoid variables mixing between two pieces of code. 

17 

18 :param original_code: code to modify 

19 :param insert_code: code to insert 

20 :param insert_code_obj: bytes sequence of inserted code, which should be modified too 

21 :param attribute_name: name of attribute to modify ('co_names', 'co_consts' or 'co_varnames') 

22 :param op_list: sequence of bytecodes whose arguments should be changed 

23 :return: modified bytes sequence of the code to insert and new values of the attribute `attribute_name` for 

24 original code 

25 """ 

26 orig_value = getattr(original_code, attribute_name) 

27 insert_value = getattr(insert_code, attribute_name) 

28 orig_names_len = len(orig_value) 

29 code_with_new_values = list(insert_code_list) 

30 offset = 0 

31 while offset < len(code_with_new_values): 

32 op = code_with_new_values[offset] 

33 if op in op_list: 

34 new_val = code_with_new_values[offset + 1] + orig_names_len 

35 if new_val > MAX_BYTE: 

36 code_with_new_values[offset + 1] = new_val & MAX_BYTE 

37 code_with_new_values = code_with_new_values[:offset] + [EXTENDED_ARG, new_val >> 8] + \ 

38 code_with_new_values[offset:] 

39 offset += 2 

40 else: 

41 code_with_new_values[offset + 1] = new_val 

42 offset += 2 

43 new_values = orig_value + insert_value 

44 return bytes(code_with_new_values), new_values

47def _modify_new_lines(code_to_modify, offset, code_to_insert): 

48 """ 

49 Update new lines: the bytecode inserted should be the last instruction of the previous line. 

50 :return: bytes sequence of code with updated lines offsets 

51 """ 

52 # There's a nice overview of co_lnotab in 

53 # https://github.com/python/cpython/blob/3.6/Objects/lnotab_notes.txt 

54 

55 if code_to_modify.co_firstlineno == 1 and offset == 0 and code_to_modify.co_name == '<module>': 

56 # There's a peculiarity here: if a breakpoint is added in the first line of a module, we 

57 # can't replace the code because we require a line event to stop and the live event 

58 # was already generated, so, fallback to tracing. 

59 return None 

60 

61 new_list = list(code_to_modify.co_lnotab) 

62 if not new_list: 

63 # Could happen on a lambda (in this case, a breakpoint in the lambda should fallback to 

64 # tracing). 

65 return None 

66 

67 # As all numbers are relative, what we want is to hide the code we inserted in the previous line 

68 # (it should be the last thing right before we increment the line so that we have a line event 

69 # right after the inserted code). 

70 bytecode_delta = len(code_to_insert) 

71 

72 byte_increments = code_to_modify.co_lnotab[0::2] 

73 line_increments = code_to_modify.co_lnotab[1::2] 

74 

75 if offset == 0: 

76 new_list[0] += bytecode_delta 

77 else: 

78 addr = 0 

79 it = zip(byte_increments, line_increments) 

80 for i, (byte_incr, _line_incr) in enumerate(it): 

81 addr += byte_incr 

82 if addr == offset: 

83 new_list[i * 2] += bytecode_delta 

84 break 

85 

86 return bytes(new_list)

89def _unpack_opargs(code, inserted_code_list, current_index): 

90 """ 

91 Modified version of `_unpack_opargs` function from module `dis`. 

92 We have to use it, because sometimes code can be in an inconsistent state: if EXTENDED_ARG 

93 operator was introduced into the code, but it hasn't been inserted into `code_list` yet. 

94 In this case we can't use standard `_unpack_opargs` and we should check whether there are 

95 some new operators in `inserted_code_list`. 

96 """ 

97 extended_arg = 0 

98 for i in range(0, len(code), 2): 

99 op = code[i] 

100 if op >= HAVE_ARGUMENT: 

101 if not extended_arg: 

102 # in case if we added EXTENDED_ARG, but haven't inserted it to the source code yet. 

103 for code_index in range(current_index, len(inserted_code_list)): 

104 inserted_offset, inserted_code = inserted_code_list[code_index] 

105 if inserted_offset == i and inserted_code[0] == EXTENDED_ARG: 

106 extended_arg = inserted_code[1] << 8 

107 arg = code[i + 1] | extended_arg 

108 extended_arg = (arg << 8) if op == EXTENDED_ARG else 0 

109 else: 

110 arg = None 

111 yield (i, op, arg)

114def _update_label_offsets(code_obj, breakpoint_offset, breakpoint_code_list): 

115 """ 

116 Update labels for the relative and absolute jump targets 

117 :param code_obj: code to modify 

118 :param breakpoint_offset: offset for the inserted code 

119 :param breakpoint_code_list: size of the inserted code 

120 :return: bytes sequence with modified labels; list of tuples (resulting offset, list of code instructions) with 

121 information about all inserted pieces of code 

122 """ 

123 inserted_code = list() 

124 # the list with all inserted pieces of code 

125 inserted_code.append((breakpoint_offset, breakpoint_code_list)) 

126 code_list = list(code_obj) 

127 j = 0 

128 

129 while j < len(inserted_code): 

130 current_offset, current_code_list = inserted_code[j] 

131 offsets_for_modification = [] 

132 

133 for offset, op, arg in _unpack_opargs(code_list, inserted_code, j): 

134 if arg is not None: 

135 if op in dis.hasjrel: 

136 # has relative jump target 

137 label = offset + 2 + arg 

138 if offset < current_offset < label: 

139 # change labels for relative jump targets if code was inserted inside 

140 offsets_for_modification.append(offset) 

141 elif op in dis.hasjabs: 

142 # change label for absolute jump if code was inserted before it 

143 if current_offset < arg: 

144 offsets_for_modification.append(offset) 

145 for i in range(0, len(code_list), 2): 

146 op = code_list[i] 

147 if i in offsets_for_modification and op >= dis.HAVE_ARGUMENT: 

148 new_arg = code_list[i + 1] + len(current_code_list) 

149 if new_arg <= MAX_BYTE: 

150 code_list[i + 1] = new_arg 

151 else: 

152 # handle bytes overflow 

153 if i - 2 > 0 and code_list[i - 2] == EXTENDED_ARG and code_list[i - 1] < MAX_BYTE: 

154 # if new argument > 255 and EXTENDED_ARG already exists we need to increase it's argument 

155 code_list[i - 1] += 1 

156 else: 

157 # if there isn't EXTENDED_ARG operator yet we have to insert the new operator 

158 extended_arg_code = [EXTENDED_ARG, new_arg >> 8] 

159 inserted_code.append((i, extended_arg_code)) 

160 code_list[i + 1] = new_arg & MAX_BYTE 

161 

162 code_list = code_list[:current_offset] + current_code_list + code_list[current_offset:] 

163 

164 for k in range(len(inserted_code)): 

165 offset, inserted_code_list = inserted_code[k] 

166 if current_offset < offset: 

167 inserted_code[k] = (offset + len(current_code_list), inserted_code_list) 

168 j += 1 

169 

170 return bytes(code_list), inserted_code

177def add_jump_instruction(jump_arg, code_to_insert): 

178 """ 

179 Note: although it's adding a POP_JUMP_IF_TRUE, it's actually no longer used now 

180 (we could only return the return and possibly the load of the 'None' before the 

181 return -- not done yet because it needs work to fix all related tests). 

182 """ 

183 extended_arg_list = [] 

184 if jump_arg > MAX_BYTE: 

185 extended_arg_list += [EXTENDED_ARG, jump_arg >> 8] 

186 jump_arg = jump_arg & MAX_BYTE 

187 

188 # remove 'RETURN_VALUE' instruction and add 'POP_JUMP_IF_TRUE' with (if needed) 'EXTENDED_ARG' 

189 return list(code_to_insert.co_code[:-RETURN_VALUE_SIZE]) + extended_arg_list + [opmap['POP_JUMP_IF_TRUE'], jump_arg]

196 # This check is needed for generator functions, because after each yield a new frame is created

202 if not ok:

215def _insert_code(code_to_modify, code_to_insert, before_line): 

216 """ 

217 Insert piece of code `code_to_insert` to `code_to_modify` right inside the line `before_line` before the 

218 instruction on this line by modifying original bytecode 

219 

220 :param code_to_modify: Code to modify 

221 :param code_to_insert: Code to insert 

222 :param before_line: Number of line for code insertion 

223 :return: boolean flag whether insertion was successful, modified code 

224 """ 

225 linestarts = dict(dis.findlinestarts(code_to_modify)) 

226 if before_line not in linestarts.values(): 

227 return False, code_to_modify 

228 

229 offset = None 

230 for off, line_no in linestarts.items(): 

231 if line_no == before_line: 

232 offset = off 

233 break 

234 

235 code_to_insert_list = add_jump_instruction(offset, code_to_insert) 

236 try: 

237 code_to_insert_list, new_names = \ 

238 _add_attr_values_from_insert_to_original(code_to_modify, code_to_insert, code_to_insert_list, 'co_names', 

239 dis.hasname) 

240 code_to_insert_list, new_consts = \ 

241 _add_attr_values_from_insert_to_original(code_to_modify, code_to_insert, code_to_insert_list, 'co_consts', 

242 [opmap['LOAD_CONST']]) 

243 code_to_insert_list, new_vars = \ 

244 _add_attr_values_from_insert_to_original(code_to_modify, code_to_insert, code_to_insert_list, 'co_varnames', 

245 dis.haslocal) 

246 new_bytes, all_inserted_code = _update_label_offsets(code_to_modify.co_code, offset, list(code_to_insert_list)) 

247 

248 new_lnotab = _modify_new_lines(code_to_modify, offset, code_to_insert_list) 

249 if new_lnotab is None: 

250 return False, code_to_modify 

251 

252 except ValueError: 

253 traceback.print_exc() 

254 return False, code_to_modify

257 code_to_modify.co_argcount, # integer 

258 code_to_modify.co_kwonlyargcount, # integer 

259 len(new_vars), # integer 

260 code_to_modify.co_stacksize, # integer 

261 code_to_modify.co_flags, # integer 

262 new_bytes, # bytes 

263 new_consts, # tuple 

264 new_names, # tuple 

265 new_vars, # tuple

267 code_to_modify.co_name, # string 

268 code_to_modify.co_firstlineno, # integer

276 args.insert(1, code_to_modify.co_posonlyargcount)
